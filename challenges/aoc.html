				file = open('input.txt', 'r')
				lines = file.readlines()
				values = [int(value.strip()) for value in lines]
				count = 0
				for i in range(0, len(values)-3):
					sum1 = values[i] + values[i+1] + values[i+2]
					sum2 = values[i+1] + values[i+2] + values[i+3]
					if(sum1<sum2):
						count += 1
				print(count)
