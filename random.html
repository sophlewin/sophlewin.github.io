                def mergeSort(numbers):
                    midpoint = len(numbers)//2
                    left_half = numbers[0:midpoint]
                    right_half = numbers[midpoint:len(numbers)]
                    
                    mergeSort(left_half)
                    mergeSort(right_half)

                    count = 0
                    left_count = 0
                    right_count = 0

                    while left_count < len(left_half) and right_count < len(right_half):
                        if right_half[right_count] > left_half[left_count]:
                            numbers[count] = right_half[right_count]
                            count+=1
                        else:
                            numbers[count] = left_half[left_count]
                            left_count+=1

                    while left_count < len(left_half):
                        numbers[count] = left_half[left_count]
                        count+=1
                        left_count+=1

                        numbers[count] = right_half[right_count]
                        count+=1
                        right_count+=1

                    return numbers
